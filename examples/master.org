* Теоретические основы
** Определения
#+name: def:graph
#+begin_definition
/Графом/ \( G=(V,E) \) называется пара: \( V \)~--- вершины, \( E\subseteq V\times V \)~--- рёбра.
#+end_definition
#+name: def:path
#+begin_definition
/Путь/ из \( s \) в \( t \): \( (v_0,\ldots,v_k) \), длина \( \ell(\pi)=\sum w(v_i,v_{i+1}) \).
#+end_definition
#+name: thm:dijkstra
#+begin_theorem
Дейкстра корректен при неотрицательных весах.
#+end_theorem
#+begin_proof
Индукция: вершина с мин. \( d[u] \) имеет окончательное значение.
#+end_proof
#+begin_corollary
Сложность: \( \compl{O}((|V|{+}|E|)\log|V|) \).
#+end_corollary
#+name: lem:triangle
#+begin_lemma
\( d(s,t)\leq d(s,u)+d(u,t) \).
#+end_lemma
#+begin_proposition
При отрицательных рёбрах Дейкстра некорректен.
#+end_proposition
#+begin_remark
Беллман--Форд корректен при отрицательных весах.
#+end_remark
#+begin_example
Дорожная сеть: вершины --- перекрёстки.
#+end_example
#+begin_problem
Все пары кратчайших путей за \( \compl{O}(|V|^3) \).
#+end_problem
Команды: \( \matr{A} \), \( \str{S}_n \), \( \elt{\sigma} \), \( \code{C} \), \( \algo{Dijkstra} \), \( \compl{O}(n\log n) \), \( \crypto{E} \).

* Алгоритмы и листинги
#+caption: Дейкстра
#+name: alg:dijkstra
\begin{algorithm}[H]
\KwInput{граф $G$, старт $s$}
\KwOutput{$d[v]$}
$d[s]\gets 0$; $Q\gets V$\tcp*{очередь}
\While{$Q\neq\emptyset$}{
    $u\gets\arg\min d[v]$\;
    \Comment{релаксация}
    \For{$(u,v)\in E$}{
        \uIf{$d[u]+w<d[v]$}{$d[v]\gets d[u]+w$\;}
        \uElseIf{равны}{обновить предка\;}
        \Else{пропустить\;}
    }
}
\Return{$d$}\;
\end{algorithm}
Ссылка: алгоритм\nbsp{}\ref{alg:dijkstra}.

\begin{lstlisting}[language=Python,caption={Python},label={lst:py}]
import heapq
def dijkstra(graph, s):
    """Дейкстра с кучей."""
    dist = {v: float('inf') for v in graph}
    dist[s] = 0
    pq = [(0, s)]
    while pq:
        d, u = heapq.heappop(pq)
        if d > dist[u]: continue
        for v, w in graph[u]:  # Соседи
            if dist[u]+w < dist[v]:
                dist[v] = dist[u]+w
                heapq.heappush(pq, (dist[v], v))
    return dist
\end{lstlisting}

\begin{lstlisting}[language=C,caption={C},label={lst:c}]
#include <stdio.h>
#include <limits.h>
/* Релаксация */
void relax(int u, int v, int w, int *d) {
    // Обновляем расстояние
    if (d[u]+w < d[v]) d[v] = d[u]+w;
}
int main(void) {
    int d[5] = {0, INT_MAX, INT_MAX, INT_MAX, INT_MAX};
    relax(0, 1, 4, d);
    printf("d[1]=%d\n", d[1]);
    return 0;
}
\end{lstlisting}

\begin{lstlisting}[style=pseudocode,caption={A*}]
(*@\lstInput@*): граф $G$, старт $s$, цель $t$, эвристика $h$
(*@\lstOutput@*): кратчайший путь
procedure AStar($G$, $s$, $t$, $h$)
begin
    $g[s]\gets 0$; $f[s]\gets h(s)$
    while очередь не пуста do
    begin
        $u\gets$ мин. $f[u]$
        if $u=t$ then return путь
        for each $(u,v)$ do
            if $g[u]+w < g[v]$ then $f[v]\gets g'+h(v)$
    end
    return nil
end
\end{lstlisting}

\begin{lstpseudocode}[caption={BFS},label={alg:bfs}]
(*@\lstInput@*): граф $G$, стартовая $s$
(*@\lstOutput@*): расстояния
function BFS($G$, $s$)
begin
    $d[s]\gets 0$; $Q\gets\{s\}$
    while $Q\neq\emptyset$ do
    begin
        $u\gets$ dequeue
        for each $(u,v)$ do
            if $d[v]=\infty$ then $d[v]\gets d[u]+1$
    end
    return $d$
end
\end{lstpseudocode}
