* Устойчивость линейных систем
** Определения
\( \vect{x}(t)\in\mathbb{R}^n \), \( \matr{A}\in\mathbb{F}^{k\times n}_q \), \( \str{S}_n \), \( \elt{\sigma} \), \( \code{C} \), \( \algo{LMI} \), \( \compl{O}(n^{3.5}) \), \( \crypto{E} \).

#+name: def:system
#+begin_definition
/Стохастическая система/:
#+name: eq:sde
\begin{equation}
d\vect{x}=\matr{A}(\xi_t)\vect{x}\,dt+\matr{B}(\xi_t)\vect{x}\,dW.
\end{equation}
#+end_definition
#+name: def:ms
#+begin_definition
/Устойчива в среднеквадратичном/, если \( \lim_{t\to\infty}\mathbb{E}\|\vect{x}(t)\|^2=0 \).
#+end_definition
#+name: thm:lmi
#+begin_theorem
Система\nbsp{}\eqref{eq:sde} экспоненциально устойчива \( \Leftrightarrow \exists\matr{P}_i\succ 0 \):
\[\matr{A}_i^\top\matr{P}_i+\matr{P}_i\matr{A}_i+\matr{B}_i^\top\matr{P}_i\matr{B}_i+\sum_j q_{ij}\matr{P}_j\prec 0.\]
#+end_theorem
#+begin_proof
Ляпунов \( V=\vect{x}^\top\matr{P}_i\vect{x} \) + формула Ито.
#+end_proof
#+begin_corollary
LMI проверяется за \( \compl{O}(N\cdot n^{3.5}) \).
#+end_corollary
#+name: lem:lyapunov
#+begin_lemma
\( \mathcal{L}V\leq-\alpha V \Rightarrow \mathbb{E}V(t)\leq e^{-\alpha t}V(0) \).
#+end_lemma
#+begin_proposition
Устойчивость \( \Rightarrow \operatorname{Re}\lambda_i(\matr{A}_j)<0 \) для некоторого \( j \).
#+end_proposition
#+begin_remark
Устойчивость подсистем не гарантирует устойчивости системы.
#+end_remark
#+begin_example
Двухрежимная: \( \matr{A}_1=\begin{pmatrix}-1&0\\0&-2\end{pmatrix} \), \( \matr{A}_2=\begin{pmatrix}-0.5&1\\-1&-0.5\end{pmatrix} \).
#+end_example
#+begin_problem
Обобщить теорему\nbsp{}[[thm:lmi]] на системы с запаздыванием.
#+end_problem

\begin{algorithm}[H]
\KwInput{$\matr{A}_i,\matr{B}_i,\matr{D}_i$, $\matr{Q}$}
\KwOutput{$\matr{K}_i$ или «невозможно»}
Сформировать LMI\;
Решить SDP\tcp*{полуопределённое}
\Comment{проверяем}
\uIf{разрешимо}{
    \For{$i=1$ \KwTo $N$}{$\matr{K}_i\gets\matr{Y}_i\matr{X}_i^{-1}$\;}
    \Return{$\matr{K}_1,\ldots,\matr{K}_N$}\;
}
\uElseIf{граница}{\Return{на границе}\;}
\Else{\Return{невозможно}\;}
\end{algorithm}

\begin{lstlisting}[language=Python,caption={Python},label={lst:py}]
import numpy as np
from scipy.linalg import solve_lyapunov
def check(A):
    """Проверка устойчивости."""
    n = A[0].shape[0]
    P = solve_lyapunov(A[0].T, -np.eye(n))
    # Проверяем положительную определённость
    return np.all(np.linalg.eigvals(P) > 0), P
A = [np.array([[-1, 0.5], [-0.5, -2]])]
ok, P = check(A)
print("Устойчива:", ok)
\end{lstlisting}

#include <stdio.h>
#include <math.h>
/* Моделирование */
void sim(double *x, double A[2][2], double dt, int n) {
    // Эйлер--Маруяма
    for (int i=0; i<n; ++i) {
        double d0=A[0][0]*x[0]+A[0][1]*x[1];
        double d1=A[1][0]*x[0]+A[1][1]*x[1];
        x[0]+=d0*dt; x[1]+=d1*dt;
    }
}
int main(void) {
    double x[2]={1,0.5}, A[2][2]={{-1,0},{0,-2}};
    sim(x,A,0.01,1000);
    printf("x=(%.4f,%.4f)\n",x[0],x[1]);return 0;
}

\begin{lstpseudocode}[caption={Монте-Карло},label={alg:mc}]
(*@\lstInput@*): система, $M$, $T$
(*@\lstOutput@*): $\mathbb{E}\|\vect{x}(T)\|^2$
function MC($M$, $T$)
begin
    $S\gets 0$
    for $j=1$ to $M$ do
        Смоделировать $\vect{x}^{(j)}(T)$
        $S\gets S+\|\vect{x}^{(j)}\|^2$
    return $S/M$
end
\end{lstpseudocode}
